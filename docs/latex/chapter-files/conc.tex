\chapter{Conclusions}\label{chap-conc}
\thispagestyle{empty}

This chapter summarizes the outcomes of the developed cold plasma generation system and highlights key observations, limitations, and future improvements.

\vspace{0.5cm}

\section{Key Contributions and Findings}

\subsection{Work Done}

\begin{itemize}
    \item Designed and implemented a \textbf{UCC21330-based H-bridge inverter} operating at \textbf{100~kHz}.
    
    \item Developed a \textbf{high-frequency step-up transformer} integrated with a \textbf{Cockcroft--Walton multiplier} for high-voltage generation.
    
    \item Implemented an \textbf{argon-based plasma generation setup} using a dielectric discharge configuration with a glass tube, high-voltage electrode, and ground ring.
    
    \item Developed an \textbf{STM32G431-based control system} for PWM generation, real-time monitoring, and overcurrent protection.
    
    \item Designed an \textbf{isolated sensing module} for safe measurement of voltage and current in the high-voltage system.
\end{itemize}
\newpage
\subsection{Key Findings}

\begin{itemize}
    \item The system successfully generated \textbf{high-voltage DC} levels of approximately \(\approx 30~\text{kV}\).
    
    \item A \textbf{stable argon plasma jet} was achieved under atmospheric conditions.
    
    \item The inverter exhibited \textbf{reliable high-frequency switching} with manageable parasitic effects.
    
    \item The voltage multiplier demonstrated \textbf{effective voltage scaling} suitable for plasma generation.
    
    \item The integrated system showed \textbf{stable and repeatable operation} under controlled conditions.
\end{itemize}

\vspace{0.3cm}


\section{Limitations}

\begin{itemize}
    
    \item \textbf{Limited output capability} of the transformer and voltage multiplier affecting plasma intensity.
    
    \item Absence of \textbf{direct high-voltage measurement}, requiring indirect estimation.
    
    \item \textbf{Thermal stress} in power components during continuous operation.
    
    \item Requirement of strict \textbf{high-voltage safety precautions}.
\end{itemize}

\vspace{0.3cm}

\section{Future Work}

\begin{itemize}
    \item Optimization of \textbf{transformer design} for higher power and efficiency.
    
    \item Development of a \textbf{closed-loop controlled high-voltage power supply}.
    
    \item Integration of \textbf{advanced diagnostics} (HV probes, current sensing, plasma analysis).
    
    \item Investigation of \textbf{electrode geometry, ground ring design, and dielectric materials}.
    
    \item Study of \textbf{different gases and flow rates} for improved plasma characteristics.
    
    \item Design of a \textbf{compact, enclosed, and safe system} with insulation and shielding.
\end{itemize}
